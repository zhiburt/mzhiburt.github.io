Interesting rustc traitbound issue with a match/if

Agenda

Here we will explore some strange behavior when calling function in a match/if statement.

Base

There's such a code example creating a HTTP router using tokio-rs/axum. It works perfectly.


    use axum::{
        extract::{Request, State},
        middleware::{from_fn_with_state, Next},
        response::Response,
        routing::get,
        Router,
    };

    fn main() {
        let middleware = from_fn_with_state((), check);
        let method = get(handler).route_layer(middleware);

        let _router: Router<()> = Router::new().route("/", method).with_state(());
    }

    async fn handler() -> &'static str {
        ""
    }

    async fn check(_state: State<()>, req: Request, next: Next) -> Response {
        next.run(req).await
    }

    fn get_some_string(_req: &Request) -> Option<String> {
        None
    }

Cases

Not lets change check function a bit, and see the issue.

Example Builds 
async fn check(_state: State<()>, req: Request, next: Next) -> Response {
    let string = get_some_string(&req);
    match string {
        Some(_) => next.run(req).await,
        None => next.run(req).await,
    }
}

true 

async fn check(_state: State<()>, req: Request, next: Next) -> Response {
    match get_some_string(&req) {
        Some(_) => next.run(req).await,
        None => next.run(req).await,
    }
}

false 

async fn check(_state: State<()>, req: Request, next: Next) -> Response {
    if let Some(_) = get_some_string(&req) {
        next.run(req).await
    } else {
        next.run(req).await
    }
}

false 

async fn check(_state: State<()>, req: Request, next: Next) -> Response {
    if let Some(_) = get_some_string(&req) {
        return next.run(req).await;
    }

    Response::default()
}

false 

async fn check(_state: State<()>, req: Request, next: Next) -> Response {
    if let Some(_) = get_some_string(&req) {
        return Response::default();
    }

    next.run(req).await
}
